11. DISK MANAGEMENT

Disk Information

List block devices


lsblk                                                           # Tree view of block devices
lsblk -f                                                        # Show filesystems
lsblk -a                                                        # Show all devices
lsblk -o NAME,SIZE,TYPE,MOUNTPOINT                              # Custom columns

Show device IDs


blkid                                                           # Show all block device attributes
blkid /dev/sda1                                                 # Show specific device
blkid -o list                                                   # List format
blkid -t TYPE=ext4                                              # Find by filesystem type
blkid -U UUID                                                   # Find by UUID

Disk usage


df                                                              # Disk space usage
df -h                                                           # Human-readable sizes
df -i                                                           # Inode usage
df -T                                                           # Show filesystem type
df -t ext4                                                      # Show only ext4 filesystems
df -x tmpfs                                                     # Exclude tmpfs
df /path                                                        # Show filesystem for path

du                                                              # Directory usage
du -h                                                           # Human-readable
du -s directory                                                 # Summary only
du -sh directory                                                # Summary human-readable
du -ah directory                                                # All files human-readable
du -h --max-depth=1                                             # Limit depth
du -ch directory                                                # Show total
du -h --exclude="*.log"                                         # Exclude pattern

Find large files


find / -type f -size +100M                                      # Files over 100MB
find / -type f -size +1G                                        # Files over 1GB
du -ah / 2>/dev/null | sort -rh | head -20                      # Top 20 largest

Partitioning

fdisk (MBR partitions)


fdisk -l                                                        # List all partitions
fdisk -l /dev/sda                                               # List partitions on device
fdisk /dev/sdb                                                  # Partition device (interactive)
# Inside fdisk:
# m - help
# p - print partition table
# n - new partition
# d - delete partition
# t - change partition type
# w - write changes and exit
# q - quit without saving

parted (GPT/MBR partitions)


parted -l                                                       # List all partitions
parted /dev/sdb                                                 # Partition device (interactive)
parted /dev/sdb print                                           # Print partition table
parted /dev/sdb mklabel gpt                                     # Create GPT partition table
parted /dev/sdb mklabel msdos                                   # Create MBR partition table
parted /dev/sdb mkpart primary ext4 0% 100%                     # Create partition
parted /dev/sdb rm 1                                            # Remove partition 1

gdisk (GPT only)


gdisk -l /dev/sda                                               # List GPT partitions
gdisk /dev/sdb                                                  # Partition device (interactive)
# Inside gdisk:
# ? - help
# p - print partition table
# n - new partition
# d - delete partition
# w - write and exit
# q - quit without saving

LVM (Logical Volume Manager)

Physical Volumes (PV)


pvs                                                             # List PVs (summary)
pvdisplay                                                       # List PVs (detailed)
pvscan                                                          # Scan for PVs
pvcreate /dev/sdb1                                              # Create PV
pvcreate /dev/sdb1 /dev/sdc1                                    # Create multiple PVs
pvremove /dev/sdb1                                              # Remove PV

Volume Groups (VG)


vgs                                                             # List VGs (summary)
vgdisplay                                                       # List VGs (detailed)
vgscan                                                          # Scan for VGs
vgcreate vg_name /dev/sdb1                                      # Create VG
vgcreate vg_name /dev/sdb1 /dev/sdc1                            # VG from multiple PVs
vgextend vg_name /dev/sdd1                                      # Add PV to VG
vgreduce vg_name /dev/sdb1                                      # Remove PV from VG
vgremove vg_name                                                # Remove VG
vgrename old_name new_name                                      # Rename VG

Logical Volumes (LV)


lvs                                                             # List LVs (summary)
lvdisplay                                                       # List LVs (detailed)
lvscan                                                          # Scan for LVs
lvcreate -n lv_name -L 10G vg_name                              # Create 10GB LV
lvcreate -n lv_name -l 100%FREE vg_name                         # Use all free space
lvcreate -n lv_name -L 10G -s /dev/vg_name/lv_source            # Create snapshot
lvextend -L +5G /dev/vg_name/lv_name                            # Extend by 5GB
lvextend -L 20G /dev/vg_name/lv_name                            # Extend to 20GB
lvextend -l +100%FREE /dev/vg_name/lv_name                      # Use all free space
lvreduce -L -5G /dev/vg_name/lv_name                            # Reduce by 5GB (DANGEROUS)
lvremove /dev/vg_name/lv_name                                   # Remove LV
lvrename vg_name old_lv new_lv                                  # Rename LV
lvresize -L 15G /dev/vg_name/lv_name                            # Resize to 15GB

Filesystems

Create filesystems

mkfs.ext4 -L label /dev/sdb1                                    # With label
mkfs.xfs /dev/sdb1                                              # Create XFS filesystem
mkfs.xfs -L label /dev/sdb1                                     # With label
mkfs.vfat /dev/sdb1                                             # Create FAT filesystem
mkfs -t ext4 /dev/sdb1                                          # Alternative syntax

Filesystem checks


fsck /dev/sdb1                                                  # Check filesystem (unmounted)
fsck -y /dev/sdb1                                               # Auto-answer yes
fsck.ext4 /dev/sdb1                                             # Check ext4
fsck.xfs /dev/sdb1                                              # Check XFS (limited)
e2fsck /dev/sdb1                                                # Check ext2/3/4

Resize filesystems


resize2fs /dev/sdb1                                             # Resize ext2/3/4 to fill partition
resize2fs /dev/sdb1 10G                                         # Resize to 10GB
xfs_growfs /mount/point                                         # Grow XFS (cannot shrink)

Filesystem information


tune2fs -l /dev/sdb1                                            # Show ext2/3/4 parameters
xfs_info /mount/point                                           # Show XFS parameters
xfs_admin -l /dev/sdb1                                          # Show XFS label
xfs_admin -L newlabel /dev/sdb1                                 # Set XFS label
tune2fs -L newlabel /dev/sdb1                                   # Set ext label

Filesystem labels


e2label /dev/sdb1                                               # Show ext label
e2label /dev/sdb1 newlabel                                      # Set ext label

Mounting

Mount filesystems


mount /dev/sdb1 /mnt                                            # Mount to /mnt
mount -t ext4 /dev/sdb1 /mnt                                    # Specify filesystem type
mount -o ro /dev/sdb1 /mnt                                      # Mount read-only
mount -o rw /dev/sdb1 /mnt                                      # Mount read-write
mount -o remount,ro /mnt                                        # Remount read-only
mount -o remount,rw /mnt                                        # Remount read-write
mount -o noexec /dev/sdb1 /mnt                                  # No execute permission
mount -o nosuid /dev/sdb1 /mnt                                  # No SUID
mount -o nodev /dev/sdb1 /mnt                                   # No device files
mount -a                                                        # Mount all in /etc/fstab
mount | grep sdb1                                               # Show if mounted

Unmount filesystems


umount /mnt                                                     # Unmount by mount point
umount /dev/sdb1                                                # Unmount by device
umount -l /mnt                                                  # Lazy unmount
umount -f /mnt                                                  # Force unmount

Show mounted filesystems


mount                                                           # Show all mounts
mount | column -t                                               # Formatted output
findmnt                                                         # Tree view of mounts
findmnt /dev/sdb1                                               # Find mount point
findmnt /mnt                                                    # Find device

/etc/fstab


# /etc/fstab format:
# device  mount_point  fs_type  options  dump  pass

# Examples (edit /etc/fstab):
# UUID=***  /data  ext4  defaults  0  2
# /dev/sdb1  /backup  xfs  defaults,noexec  0  2
# //server/share  /mnt/cifs  cifs  credentials=/root/.creds  0  0

# Test fstab
mount -a                                                        # Mount all (test fstab)
findmnt --verify                                                # Verify fstab syntax

# Get UUID
blkid /dev/sdb1                                                 # Show UUID
lsblk -f                                                        # Show UUIDs for all

Swap Management

Show swap


swapon --show                                                   # Show swap devices
swapon -s                                                       # Show swap summary
cat /proc/swaps                                                 # Show swap (alternative)
free -h                                                         # Show memory and swap

Create swap partition


mkswap /dev/sdb1                                                # Initialize swap partition
swapon /dev/sdb1                                                # Enable swap
swapon -p 10 /dev/sdb1                                          # Enable with priority 10

Create swap file


dd if=/dev/zero of=/swapfile bs=1M count=2048                   # Create 2GB file
chmod 600 /swapfile                                             # Secure permissions
mkswap /swapfile                                                # Initialize swap file
swapon /swapfile                                                # Enable swap file

Disable swap


swapoff /dev/sdb1                                               # Disable swap partition
swapoff -a                                                      # Disable all swap

Add to /etc/fstab for persistence:


# /dev/sdb1  none  swap  defaults  0  0
# /swapfile  none  swap  defaults  0  0

RAID (mdadm)

Create RAID


mdadm --create /dev/md0 --level=1 --raid-devices=2 /dev/sdb1 /dev/sdc1            # RAID 1
mdadm --create /dev/md0 --level=5 --raid-devices=3 /dev/sdb1 /dev/sdc1 /dev/sdd1  # RAID 5

Show RAID status


cat /proc/mdstat                                                # RAID status
mdadm --detail /dev/md0                                         # Detailed RAID info
mdadm --query /dev/md0                                          # Query RAID device

Manage RAID


mdadm --stop /dev/md0                                           # Stop RAID
mdadm --assemble /dev/md0                                       # Assemble RAID
mdadm /dev/md0 --fail /dev/sdb1                                 # Mark disk as failed
mdadm /dev/md0 --remove /dev/sdb1                               # Remove disk
mdadm /dev/md0 --add /dev/sde1                                  # Add disk

Save RAID configuration


mdadm --detail --scan >> /etc/mdadm.conf                        # Save config

Network Configuration

Show IP addresses


ip addr                                                         # Show all interfaces
ip addr show                                                    # Same as above
ip a                                                            # Short form
ip addr show eth0                                               # Show specific interface
ip -4 addr                                                      # IPv4 only
ip -6 addr                                                      # IPv6 only
ip -br addr                                                     # Brief format
ip -c addr                                                      # Colorized output

ifconfig                                                        # Show all interfaces
ifconfig eth0                                                   # Show specific interface

IP Address Management

Add IP address


ip addr add 192.168.1.10/24 dev eth0                            # Add IP
ip addr add 192.168.1.10/24 brd + dev eth0                      # With broadcast

Delete IP address


ip addr del 192.168.1.10/24 dev eth0                            # Remove IP

Flush IP addresses


ip addr flush dev eth0                                          # Remove all IPs from interface

Change MTU


ip link set eth0 mtu 9000                                       # Set MTU to 9000

Change MAC address

Legacy commands


ifconfig eth0 up                                                # Bring up

Routing

Show routing table

Change routes

arp -a                                                          # Show ARP cache

14. SSH (SECURE SHELL)

SSH Client

Basic connection


ssh user@hostname                                               # Connect to host
ssh user@192.168.1.10                                           # Connect by IP
ssh -p 2222 user@hostname                                       # Custom port
ssh hostname                                                    # Connect with current username

SSH options


ssh -v user@hostname                                            # Verbose mode
ssh -vv user@hostname                                           # More verbose
ssh -vvv user@hostname                                          # Maximum verbosity
ssh -4 user@hostname                                            # Force IPv4
ssh -6 user@hostname                                            # Force IPv6
ssh -C user@hostname                                            # Compression
ssh -X user@hostname                                            # X11 forwarding
ssh -Y user@hostname                                            # Trusted X11 forwarding

Execute remote command


ssh user@hostname 'command'                                     # Execute single command
ssh user@hostname 'ls -la'                                      # Example
ssh user@hostname 'uname -a; df -h'                             # Multiple commands

SSH with key


ssh -i ~/.ssh/id_rsa user@hostname                              # Specify private key

SSH config usage


ssh confighost                                                  # Use ~/.ssh/config entry

SSH Key Management

Generate SSH keys


ssh-keygen                                                      # Generate RSA key (default)
ssh-keygen -t rsa                                               # RSA key explicitly
ssh-keygen -t rsa -b 4096                                       # 4096-bit RSA
ssh-keygen -t ecdsa                                             # ECDSA key
ssh-keygen -t ed25519                                           # Ed25519 key (recommended)
ssh-keygen -t rsa -C "comment"                                  # With comment
ssh-keygen -f ~/.ssh/custom_key                                 # Custom filename
ssh-keygen -N ""                                                # No passphrase
ssh-keygen -p                                                   # Change passphrase

Copy public key to server


ssh-copy-id user@hostname                                       # Copy default key
ssh-copy-id -i ~/.ssh/id_rsa.pub user@hostname                  # Specify key
ssh-copy-id -p 2222 user@hostname                               # Custom port

Manual key copy


cat ~/.ssh/id_rsa.pub | ssh user@hostname "mkdir -p ~/.ssh && cat >> ~/.ssh/authorized_keys"

Key permissions


chmod 700 ~/.ssh                                                # SSH directory
chmod 600 ~/.ssh/id_rsa                                         # Private key
chmod 644 ~/.ssh/id_rsa.pub                                     # Public key
chmod 644 ~/.ssh/authorized_keys                                # Authorized keys
chmod 600 ~/.ssh/config                                         # SSH config
