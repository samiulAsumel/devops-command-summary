4. FILE PERMISSIONS & OWNERSHIP
Understanding Permissions

# View permissions
ls -l                       # Long format shows permissions
ls -la                      # Include hidden files
stat file.txt               # Detailed file status including permissions

# Permission format: -rwxrwxrwx
# Position 1: file type (- file, d directory, l link)
# Positions 2-4: owner (user) permissions
# Positions 5-7: group permissions
# Positions 8-10: others permissions
# r=read(4), w=write(2), x=execute(1)
Change Permissions (chmod)

# Symbolic mode
chmod u+x file              # Add execute for user
chmod g-w file              # Remove write for group
chmod o+r file              # Add read for others
chmod a+x file              # Add execute for all
chmod u+rw,g+r file         # Multiple changes
chmod ugo+x file            # Execute for user, group, others

# Numeric (octal) mode
chmod 644 file              # rw-r--r-- (common for files)
chmod 755 file              # rwxr-xr-x (common for scripts)
chmod 777 file              # rwxrwxrwx (DANGEROUS - avoid)
chmod 600 file              # rw------- (private file)
chmod 700 directory         # rwx------ (private directory)

# Recursive
chmod -R 755 directory      # Apply to directory and contents
chmod -R u+x directory      # Add execute recursively

# Special permissions
chmod u+s file              # Set SUID (setuid)
chmod g+s directory         # Set SGID (setgid)
chmod +t directory          # Set sticky bit
chmod 4755 file             # SUID + rwxr-xr-x
chmod 2755 directory        # SGID + rwxr-xr-x
chmod 1777 directory        # Sticky + rwxrwxrwx (like /tmp)
Change Ownership (chown)

# Change owner
chown newowner file         # Change file owner
chown -R newowner dir       # Recursive

# Change owner and group
chown user:group file       # Change both
chown user: file            # Change user, group to user's primary group

# Change only group (alternative to chgrp)
chown :group file           # Change group only
Change Group (chgrp)

chgrp newgroup file         # Change group
chgrp -R newgroup dir       # Recursive
Access Control Lists (ACL)

# View ACLs
getfacl file.txt            # Show ACL for file
getfacl -R directory        # Recursive

# Set ACLs
setfacl -m u:username:rwx file     # Modify user ACL
setfacl -m g:groupname:rx file     # Modify group ACL
setfacl -m o::r file               # Modify others ACL
setfacl -x u:username file         # Remove user ACL
setfacl -b file                    # Remove all ACLs
setfacl -R -m u:username:rx dir    # Recursive
setfacl -d -m u:username:rwx dir   # Set default ACL for new files

# Copy ACLs
getfacl file1 | setfacl --set-file=- file2  # Copy ACL from file1 to file2
File Attributes (Extended)

# View attributes
lsattr file.txt             # Show file attributes
lsattr -a directory         # Include hidden files
lsattr -d directory         # Show directory attributes

# Change attributes
chattr +i file              # Immutable (cannot modify/delete)
chattr -i file              # Remove immutable
chattr +a file              # Append only
chattr +A file              # No atime updates
chattr +c file              # Compressed
chattr +s file              # Secure deletion
chattr +u file              # Undeletable
umask (Default Permissions)

umask                       # Show current umask
umask 022                   # Set umask (files: 644, dirs: 755)
umask 027                   # Set umask (files: 640, dirs: 750)
umask -S                    # Show in symbolic format

# Make permanent (add to ~/.rc or /etc/profile)
5. USER & GROUP MANAGEMENT
User Operations

# Create users
useradd username                        # Create user (minimal setup)
useradd -m username                     # Create with home directory
useradd -m -s /bin/ username            # Specify shell
useradd -m -d /custom/home username     # Custom home directory
useradd -m -u 1500 username             # Specify UID
useradd -m -g group username            # Specify primary group
useradd -m -G group1,group2 username    # Add to groups
useradd -m -e 2024-12-31 username       # Expiry date
useradd -m -c "Full Name" username      # Comment (GECOS)

# Modify users
usermod -s /bin/zsh username      # Change shell
usermod -d /new/home username     # Change home directory
usermod -m -d /new/home username  # Move home directory contents
usermod -l newname oldname        # Change username
usermod -L username               # Lock account
usermod -U username               # Unlock account
usermod -e 2024-12-31 username    # Set expiry date
usermod -e "" username            # Remove expiry date
usermod -g newgroup username      # Change primary group
usermod -G group1,group2 username # Set supplementary groups (REPLACE)
usermod -aG group username        # Add to group (APPEND)

# Delete users
userdel username            # Delete user (keep home directory)
userdel -r username         # Delete user and home directory
userdel -f username         # Force delete (even if logged in)

# Password management
passwd username             # Set/change password
passwd -l username          # Lock password
passwd -u username          # Unlock password
passwd -d username          # Delete password (passwordless)
passwd -e username          # Expire password (force change)
passwd -S username          # Show password status
passwd -n 7 username        # Minimum password age (7 days)
passwd -x 90 username       # Maximum password age (90 days)
passwd -w 14 username       # Warning days before expiry

# Password aging
chage username                  # Interactive password aging
chage -l username               # List aging information
chage -E 2024-12-31 username    # Set account expiry
chage -M 90 username            # Max days before password change
chage -m 7 username             # Min days between password changes
chage -W 14 username            # Warning days before expiry
chage -I 30 username            # Inactive days after expiry

Interactive Process Monitoring

# top
top                         # Interactive process viewer

# Press M - sort by memory
# Press P - sort by CPU
# Press k - kill process (enter PID)
# Press r - renice process
# Press f - select fields to display
# Press q - quit

top -d 5                    # Update every 5 seconds
top -b -n 1 > output.txt    # Batch mode, one iteration

# htop (if installed)
htop                        # Interactive process viewer
# F5 - tree view
# F6 - sort
# F9 - kill
# F10 - quit

# atop (comprehensive resource monitor)
atop                        # System and process monitor

pkill -u username           # Kill by user
pkill -f pattern            # Match full command line

# Send signals

# Substring

# String replacement

echo ${string//o/O}             # Replace all 'o' with 'O'
echo ${string/#Hello/Hi}        # Replace at beginning
echo ${string/%World/Universe}  # Replace at end

# Remove pattern
echo ${string#He}            # Remove shortest match from beginning
echo ${string##He*o}         # Remove longest match from beginning
echo ${string%ld}            # Remove shortest match from end
echo ${string%%o*}           # Remove longest match from end

# Case conversion
echo ${string^^}             # Uppercase all
echo ${string,,}             # Lowercase all
echo ${string^}              # Uppercase first character
echo ${string,}              # Lowercase first character

# Default values
echo ${var:-default}         # Use default if var is unset or null
echo ${var:=default}         # Assign default if var is unset or null
echo ${var:+value}           # Use value if var is set
echo ${var:?error message}   # Print error if var is unset or null

history (Command History)

# View history
history                      # Show all history
history 20                   # Show last 20 commands
history | grep ssh           # Search history

# Execute from history
!100                         # Execute command 100
!!                           # Execute last command
!-2                          # Execute 2 commands ago
!ssh                         # Execute last command starting with ssh
!?pattern                    # Execute last command containing pattern

# History control
history -c                          # Clear history
history -d 100                      # Delete command 100
export HISTSIZE=10000               # Set history size
export HISTFILESIZE=20000           # Set history file size
export HISTCONTROL=ignoredups       # Ignore duplicates
export HISTIGNORE="ls:pwd:exit"     # Ignore specific commands

# Search history
Ctrl+r                       # Reverse search (interactive)
