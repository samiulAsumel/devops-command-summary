LVM (Logical Volume Manager)

Understanding LVM Concepts


# LVM Structure (bottom to top):
# Physical Volumes (PV) - Raw disks/partitions
#   ↓
# Volume Groups (VG) - Pool of PVs
#   ↓
# Logical Volumes (LV) - Virtual partitions

# Why use LVM?
# 1. Flexible resizing - Extend/reduce volumes on-the-fly
# 2. Snapshots - Point-in-time backups
# 3. Striping - Spread data across disks (performance)
# 4. Mirroring - Data redundancy
# 5. Migration - Move data between disks without downtime

# LVM Workflow Example:
# 1. Create PV: pvcreate /dev/sdb1
# 2. Create VG: vgcreate myvg /dev/sdb1
# 3. Create LV: lvcreate -L 10G -n mylv myvg
# 4. Create filesystem: mkfs.ext4 /dev/myvg/mylv
# 5. Mount: mount /dev/myvg/mylv /mnt

# Important Notes:
# - Always unmount before reducing LV size (data loss risk)
# - Extending is safer than reducing
# - Snapshots consume space as changes accumulate
# - Remove snapshots when no longer needed

# - D-Bus interface for services
# - Runtime and permanent rule sets
# - Easier to manage than raw iptables

# iptables (Legacy)
# - Table-based firewall (filter, nat, mangle)
# - Static (requires restart for rule changes)
# - More granular control
# - Steeper learning curve

# Firewalld Zone Concepts:
# Zone = Set of rules defining trust level
#
# Common Zones (least to most trusted):
# drop - Drop all incoming, no reply
# block - Reject all incoming with icmp-host-prohibited
# public - Public networks (default) - allow specific services
# external - External network with masquerading
# dmz - DMZ (demilitarized zone) - limited access
# work - Work environment - more trusted
# home - Home network - even more trusted
# internal - Internal network
# trusted - Accept all traffic

# Rule Precedence:
# 1. Direct rules (raw iptables)
# 2. Rich rules (advanced filtering)
# 3. Service rules (predefined services)
# 4. Port rules (simple port opening)

# Temporary vs Permanent:
# Temporary: Active until reload/reboot
#   firewall-cmd --add-service=http
# Permanent: Survives reload/reboot
#   firewall-cmd --permanent --add-service=http
#   firewall-cmd --reload

# Best Practices:
# - Always use --permanent for production rules
# - Test rules without --permanent first

# - Use services instead of raw ports when possible

SSH Key Management

Understanding SSH Key Authentication


# Why SSH Keys over Passwords?
# 1. More secure - 2048+ bit encryption vs weak passwords
# 2. Automation - Scripts can connect without manual input
# 3. No password transmission - Keys never sent over network
# 4. Revocable - Remove public key to deny access

# Key Types Comparison:
# RSA (default, widely compatible)
#   - 2048-bit minimum, 4096-bit recommended
#   - ssh-keygen -t rsa -b 4096
#
# Ed25519 (modern, recommended)
#   - Faster, more secure than RSA
#   - Smaller key size (256-bit = RSA 3072-bit security)
#   - ssh-keygen -t ed25519
#
# ECDSA (elliptic curve)
#   - Faster than RSA, less common
#   - ssh-keygen -t ecdsa -b 521

# SSH Key Pair:
# Private Key (~/.ssh/id_rsa) - Keep SECRET, never share
# Public Key (~/.ssh/id_rsa.pub) - Can be shared freely

# Authentication Flow:
# 1. Client sends public key fingerprint
# 2. Server checks ~/.ssh/authorized_keys
# 3. Server sends encrypted challenge
# 4. Client decrypts with private key
# 5. Client sends response
# 6. Server grants access if correct

# Security Best Practices:
# - Always use passphrase for private keys
# - Use ssh-agent to avoid re-entering passphrase
# - Disable password authentication after key setup
#   (edit /etc/ssh/sshd_config: PasswordAuthentication no)
# - Restrict private key permissions: chmod 600 ~/.ssh/id_rsa
# - Regularly rotate keys (especially for service accounts)

🚀 Chapter 2: YAML Basic Syntax and Rules


# YAML Syntax Rules (Must Follow!)

# 1. Indentation (Most Important Rule)
# Use spaces, NOT tabs
# Consistent indentation level
# 2 spaces or 4 spaces (be consistent)

# ✅ CORRECT:
database:
  host: localhost
  port: 5432
  name: myapp

# ❌ WRONG (mixed tabs and spaces):
database:
	host: localhost		# Tab character
  port: 5432
	name: myapp

# 2. Key-Value Pairs
# Basic structure: key: value
# Space after colon is required

# ✅ CORRECT:
name: John Doe
age: 25
city: Dhaka
is_active: true

# ❌ WRONG (no space after colon):
name:John Doe		# Missing space
age:25		# Missing space

# 3. Comments
# Use # for comments
# Comments must be at beginning of line or after space

# This is a full line comment
name: John  # This is an inline comment
age: 25     # Another inline comment

# 4. Strings (Text Data)
# Can use quotes or not (recommended to use quotes)

# ✅ CORRECT:
name: "John Doe"
description: 'This is a description'
simple_text: This works too

# Special characters in strings need quotes:
special: "Hello, World!"    # Comma needs quotes
with_colon: "Time: 10:30"  # Colon needs quotes

# 5. Numbers
# Integers and floats (no quotes needed)

count: 42
temperature: 25.5
percentage: 0.75

# Scientific notation:
large_number: 1.23e+4
small_number: 1.23e-4

# 6. Booleans
# true/false, yes/no, on/off (case insensitive)

is_active: true
is_admin: false
has_access: yes
is_enabled: no
debug_mode: on
production: off

# 7. Null Values
# Represent empty/no value

empty_field: null
no_data: ~
optional_field: null

# 8. Lists/Arrays
# Use hyphen (-) with space

# ✅ CORRECT:
fruits:
  - apple
  - banana
  - orange
  - mango

# Inline list (compact):
fruits: [apple, banana, orange, mango]

# List of objects:
users:
  - name: John
    age: 25
    city: Dhaka
  - name: Alice
    age: 30
    city: Chittagong

# 9. Dictionaries/Objects
# Key-value pairs under a parent

# ✅ CORRECT:
person:
  name: John Doe
  age: 25
  address:
    street: "123 Main St"
    city: Dhaka
    country: Bangladesh
  contacts:
    email: "john@example.com"
    phone: "+8801712345678"

# 10. Multi-line Strings
# Use | (literal) or > (folded)

# Literal block (preserves newlines):
description: |
  This is a multi-line
  string that preserves
  all line breaks exactly
  as written here.

# Folded block (converts newlines to spaces):
description: >
  This is a multi-line
  string that gets folded
  into a single line
  with spaces instead.

# 11. Advanced Data Types
# Dates, timestamps, binary data

created_at: 2025-01-13
updated_at: 2025-01-13T19:30:00Z
binary_data: !!binary R0lGODlhDA...

📝 Chapter 3: YAML Advanced Features


# Advanced YAML Features

# 1. Anchors and Aliases (Reuse Content)
# Define anchor with &, reference with *

# ✅ CORRECT:
defaults: &defaults
  timeout: 30
  retries: 3
  debug: false

database:
  <<: *defaults    # Merge all defaults
  host: localhost
  port: 5432

cache:
  <<: *defaults    # Merge all defaults
  host: redis-server
  port: 6379

# 2. Override with Anchors
# Override specific values

database:
  <<: *defaults
  host: localhost
  port: 5432
  timeout: 60      # Override default timeout

# 3. Multiple Anchors
base_config: &base
  timeout: 30
  retries: 3

security_config: &security
  ssl: true
  encryption: AES256

# Merge multiple anchors:
production_config:
  <<: [*base, *security]
  host: prod.example.com
  port: 443

# 4. Tags and Custom Types
# Specify data types explicitly

# Date parsing:
created_at: !!str 2025-01-13  # Force as string
birth_date: !!timestamp 1990-01-01

# Binary data:
image_data: !!binary R0lGODlhDA...

# Custom types:
version: !!float 1.0
precision: !!int 42

# 5. Merge Keys (<<)
# Merge multiple dictionaries

base: &base
  common_setting: value
  another_setting: another

override:
  common_setting: overridden_value

final:
  <<: [*base, *override]
  # Result:
  # common_setting: overridden_value
  # another_setting: another

# 6. Flow Scalars (Complex Strings)
# Use { } for inline JSON-like structures

# Complex inline data:
complex_data: {name: John, age: 25, active: true}
inline_list: [apple, banana, orange]

# Mixed with regular YAML:
user:
  name: "John Doe"
  metadata: {role: admin, department: IT}
  tags: [developer, senior, full-time]

# 7. Document Separators
# Use --- to separate multiple documents

---
document: 1
data: First document

---
document: 2
data: Second document

---
document: 3
data: Third document

# 8. Directives and Tags
# %TAG, %YAML directives

%TAG ! tag:example.com,2000:
%YAML 1.2
---
custom_tagged: !custom value

# Calculator Function:
function calculator() {
    local num1=$1
    local operator=$2
    local num2=$3
    
    case $operator in
        "+")
            result=$((num1 + num2))
            ;;
        "-")
            result=$((num1 - num2))
            ;;
        "*")
            result=$((num1 * num2))
            ;;
        "/")
            if [ $num2 -ne 0 ]; then
                result=$(echo "scale=2; $num1 / $num2" | bc)
            else
                echo "Error: Division by zero!"
                return 1
            fi
            ;;
        *)
            echo "Error: Unknown operator $operator"
            return 1
            ;;
    esac
    
    echo "Result: $result"
}

# Secure File Operations
secure_write() {
    local content=$1
    local file=$2
    local temp_file="${file}.tmp.$$"
    
    # Write to temp file first
    echo "$content" > "$temp_file"
    
    # Verify content
    if [ ! -s "$temp_file" ]; then
        log_message "ERROR" "Failed to write to temp file"
        rm -f "$temp_file"
        return 1
    fi
    
    # Atomic move (prevents corruption)
    mv "$temp_file" "$file"
    chmod 644 "$file"
    log_message "INFO" "Secure write completed: $file"
}

# File Integrity Verification
verify_file_integrity() {
    local file=$1
    local expected_hash=$2
    
    if [ ! -f "$file" ]; then
        log_message "ERROR" "File not found: $file"
        return 1
    fi
    
    local actual_hash=$(sha256sum "$file" | cut -d' ' -f1)
    
    if [ "$actual_hash" != "$expected_hash" ]; then
        log_message "ERROR" "File integrity check failed: $file"
        return 1
    fi
    
    log_message "INFO" "File integrity verified: $file"
    return 0
}

Understanding rsync vs Other Backup Tools


# rsync vs cp:
# cp - Copies everything every time
# rsync - Only transfers changed files (incremental)
#
# rsync vs scp:
# scp - Copies entire file even if partially exists
# rsync - Resumes partial transfers, delta-transfer algorithm
#
# rsync vs tar:
# tar - Archives files into single file
# rsync - Syncs directories maintaining structure

# rsync Algorithm:
# 1. Builds file list
# 2. Compares checksums (rolling checksum)
# 3. Transfers only changed blocks
# 4. Verifies integrity after transfer

# Common rsync Modes:
# Backup mode: rsync -av --delete /source/ /backup/
#   - Archive mode (preserves permissions, times, symlinks)
#   - Delete files in destination not in source
#
# Mirror mode: rsync -av /source/ user@remote:/dest/
#   - Push local changes to remote
#
# Restore mode: rsync -av user@remote:/backup/ /restore/
#   - Pull backup from remote
#
# Incremental backup: rsync -av --link-dest=/backup/prev /source/ /backup/new/
#   - Hard links to unchanged files (saves space)

# Important Flags Explained:
# -a (archive) = -rlptgoD
# -r: recursive
# -l: copy symlinks as symlinks
# -p: preserve permissions
# -t: preserve modification times
# -g: preserve group
# -o: preserve owner (requires sudo)
# -D: preserve device/special files
# -v: verbose output
# -z: compress during transfer (slower CPU, faster network)
# -P: combine --progress and --partial
# --delete: delete extraneous files from dest

# Trailing Slash Behavior (IMPORTANT):
# rsync -av /source/ /dest/    # Contents of source → dest
# rsync -av /source /dest/     # source directory → dest/source

# Bandwidth Management:
# rsync -av --bwlimit=1000 /source/ /dest/
#   - Limits to 1000 KB/s (1 MB/s)
#   - Useful for daytime backups without impacting users

# Cron Time Field Ranges:
# Minute:     0-59
# Hour:       0-23

# 0 2 * * *        Every day at 2:00 AM
# */15 * * * *     Every 15 minutes

#!/bin/bash
# Enterprise backup script with error handling

set -euo pipefail
IFS=$'\n\t'

# Logging function
log() {
echo "[$(date +'%Y-%m-%d %H:%M:%S')] $1" | tee -a /var/log/backup.log
}

# Error handling
error_exit() {
log "ERROR: $1"
exit 1
}

# Backup function with validation
backup_directory() {
local src_dir=$1
local dest_dir=$2
local backup_file="${dest_dir}/backup_$(date +%Y%m%d_%H%M%S).tar.gz"

log "Starting backup of $src_dir"

# Create destination if not exists
mkdir -p "$dest_dir" || error_exit "Cannot create destination directory"

# Perform backup with compression
tar -czf "$backup_file" -C "$(dirname "$src_dir")" "$(basename "$src_dir")" || error_exit "Backup failed"

# Verify backup integrity
tar -tzf "$backup_file" > /dev/null || error_exit "Backup verification failed"

log "Backup completed: $backup_file"
log "Backup size: $(du -h "$backup_file" | cut -f1)"
}

# Main execution
main() {
local backup_source="/home"
local backup_dest="/backup"

# Check root privileges
[[ $EUID -eq 0 ]] || error_exit "This script must be run as root"

# Check available space
local required_space=$(du -s "$backup_source" | cut -f1)
local available_space=$(df "$backup_dest" | awk 'NR==2 {print $4*1024}')

if [[ $required_space -gt $available_space ]]; then
error_exit "Insufficient disk space for backup"
fi

backup_directory "$backup_source" "$backup_dest"

# Cleanup old backups (keep last 7)
find "$backup_dest" -name "backup_*.tar.gz" -mtime +7 -delete
log "Old backups cleaned up"
}

main "$@"

#!/bin/bash
# Comprehensive system monitoring

ALERT_THRESHOLD_MEM=90
ALERT_THRESHOLD_DISK=85
LOG_FILE="/var/log/system_monitor.log"

check_cpu() {
local cpu_usage=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1)
if (( $(echo "$cpu_usage > $ALERT_THRESHOLD_CPU" | bc -l) )); then
echo "[$(date)] CPU ALERT: ${cpu_usage}%" >> "$LOG_FILE"
return 1
fi
return 0
}

check_memory() {
local mem_usage=$(free | grep Mem | awk '{printf "%.1f", $3/$2 * 100.0}')
if (( $(echo "$mem_usage > $ALERT_THRESHOLD_MEM" | bc -l) )); then
echo "[$(date)] MEMORY ALERT: ${mem_usage}%" >> "$LOG_FILE"
return 1
fi
return 0
}

check_disk() {
local disk_usage=$(df / | awk 'NR==2 {print $5}' | cut -d'%' -f1)
if [[ $disk_usage -gt $ALERT_THRESHOLD_DISK ]]; then
echo "[$(date)] DISK ALERT: ${disk_usage}%" >> "$LOG_FILE"
return 1
fi
return 0
}

check_services() {
local services=("nginx" "mysql" "sshd")
for service in "${services[@]}"; do
if ! systemctl is-active --quiet "$service"; then
echo "[$(date)] SERVICE ALERT: $service is down" >> "$LOG_FILE"
fi
done
}

# Main monitoring loop
while true; do

check_memory
check_disk
check_services
